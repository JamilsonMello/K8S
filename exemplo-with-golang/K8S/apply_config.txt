apply configuration:
    ==> kubectl apply -f filename.yaml

list pods:
    ==> kubectl get pods
    ==> kubectl get pods
    ==> kubectl get po

set container port:
    ==> kubectl port-forward pod/goserver 8080:8080

delete pod:
    ==> kubectl delete pod <namepod>

list replicasets:
    ==> kubectl get replicasets

describe a pod:
    ==> kubectl describe pod <namepod>
